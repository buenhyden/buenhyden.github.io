# Iteration
for i in range(5):
    print(i)

# Conditionals
if x > 0:
    print("Positive")
elif x < 0:
    print("Negative")
else:
    print("Zero")

# Exception Handling
try:
    result = 10 / 0
except ZeroDivisionError:
    print("Cannot divide by zero")
finally:
    print("Cleanup")

// Iteration

    System.out.println(i);

// Conditionals

    System.out.println("Positive");
} else if (x < 0) {

    System.out.println("Zero");

// Exception Handling
try {
    int result = 10 / 0;
} catch (ArithmeticException e) {
    System.out.println("Cannot divide by zero");
} finally {
    System.out.println("Cleanup");
}

// Iteration
for (i in 0.) {
    println(i)
}

// Conditionals
when {
    x > 0 -> println("Positive")
    x < 0 -> println("Negative")
    else -> println("Zero")
}

// Exception Handling

    val result = 10 / 0
} catch (e: ArithmeticException) {
    println("Cannot divide by zero")
} finally {
    println("Cleanup")

// Iteration
for i in 0. {
    println!("{}", i);
}

// Conditionals
match x {
    x if x > 0 => println!("Positive"),
    x if x < 0 => println!("Negative"),
    _ => println!("Zero"),
}

// Error Handling
match some_function() {
    Ok(result) => println!("Success: {}", result),
    Err(error) => println!("Error: {}", error),
}
